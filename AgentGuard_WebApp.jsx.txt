function Badge({ trust, size = "sm" }) {
  const col = trustColor(trust);
  return (
    <span style={{ display: "inline-flex", alignItems: "center", gap: 4, padding: size === "lg" ? "4px 12px" : "2px 8px", fontSize: size === "lg" ? 12 : 10, fontFamily: MONO, fontWeight: 600, color: col, background: col + "12", border: `1px solid ${col}30`, borderRadius: 3 }}>
      {trustIcon(trust)} {trustLabel(trust)}
    </span>
  );
}

function ScoreBar({ score, width = 10 }) {
  const pct = score / 10;
  const col = score >= 8 ? C.green : score >= 5 ? C.amber : C.red;
  const filled = Math.round(pct * width);
  return (
    <span style={{ fontFamily: MONO, fontSize: 12 }}>
      <span style={{ color: col }}>{"[" + "█".repeat(filled) + "░".repeat(width - filled) + "]"}</span>
      <span style={{ color: col, marginLeft: 4 }}>{score.toFixed(1)}</span>
    </span>
  );
}

function PermTag({ perm }) {
  const col = riskColor(perm.r);
  const danger = perm.r === "high" || perm.r === "critical";

  );
}

// ─── DIRECTORY PAGE ───
function DirectoryPage({ setPage, skills, searchQuery }) {
  const [filter, setFilter] = useState("all");
  const [platform, setPlatform] = useState("all");
  const [sort, setSort] = useState("score");

  let filtered = skills;
  if (filter === "safe") filtered = filtered.filter(s => s.trust === "verified" || s.trust === "scanned");
  if (filter === "threats") filtered = filtered.filter(s => s.trust === "flagged");
  if (platform !== "all") filtered = filtered.filter(s => s.platform === platform);

  if (searchQuery) {
    const q = searchQuery.toLowerCase();
    filtered = filtered.filter(s => s.name.includes(q) || s.desc.toLowerCase().includes(q) || s.tags.some(t => t.includes(q)) || s.platform.includes(q) || s.author.includes(q));
  }

  if (sort === "score") filtered.sort((a, b) => b.score - a.score);
  if (sort === "downloads") filtered.sort((a, b) => b.dlWeek - a.dlWeek);
  if (sort === "newest") filtered.sort((a, b) => b.updated.localeCompare(a.updated));
  if (sort === "reviews") filtered.sort((a, b) => b.reviews - a.reviews);

  return (
    <section style={{ padding: "24px", maxWidth: 900, margin: "0 auto" }}>
      {/* Header */}
      <div style={{ fontFamily: MONO, fontSize: 12, color: C.darkGray, marginBottom: 16 }}>
        <span style={{ color: C.green }}>$</span> agentguard list {searchQuery ? `--query "${searchQuery}" ` : ""}{filter !== "all" ? `--${filter} ` : ""}{platform !== "all" ? `--platform ${platform} ` : ""}--sort {sort}
      </div>

      {/* Filters bar */}
      <div style={{ display: "flex", gap: 8, marginBottom: 16, flexWrap: "wrap", alignItems: "center" }}>
        {/* Trust filter */}
        {[{ k: "all", l: "all" }, { k: "safe", l: "--safe" }, { k: "threats", l: "--threats" }].map(f => (
          <button key={f.k} onClick={() => setFilter(f.k)} style={{ ...filterBtnStyle, background: filter === f.k ? (f.k === "threats" ? C.red + "15" : f.k === "safe" ? C.green + "15" : C.dimGray) : "transparent", color: filter === f.k ? (f.k === "threats" ? C.red : f.k === "safe" ? C.green : C.white) : C.darkGray, borderColor: filter === f.k ? (f.k === "threats" ? C.red + "40" : f.k === "safe" ? C.green + "40" : C.border) : C.border }}>{f.l}</button>
        ))}
        <span style={{ fontFamily: MONO, fontSize: 10, color: C.dimGray }}>│</span>
        {/* Platform filter */}
        <select value={platform} onChange={e => setPlatform(e.target.value)} style={{ fontFamily: MONO, fontSize: 11, background: C.surface, color: C.gray, border: `1px solid ${C.border}`, padding: "4px 8px", cursor: "pointer", outline: "none" }}>
          <option value="all">all platforms</option>
          {PLATFORMS.map(p => <option key={p.slug} value={p.slug}>{p.name}</option>)}
        </select>
        <span style={{ fontFamily: MONO, fontSize: 10, color: C.dimGray }}>│</span>
        {/* Sort */}
        <select value={sort} onChange={e => setSort(e.target.value)} style={{ fontFamily: MONO, fontSize: 11, background: C.surface, color: C.gray, border: `1px solid ${C.border}`, padding: "4px 8px", cursor: "pointer", outline: "none" }}>
          <option value="score">sort: score</option>
          <option value="downloads">sort: downloads</option>
          <option value="newest">sort: newest</option>
          <option value="reviews">sort: reviews</option>
        </select>
        <span style={{ fontFamily: MONO, fontSize: 10, color: C.darkGray, marginLeft: "auto" }}>{filtered.length} results</span>
      </div>

      {/* Results */}
      {filtered.length === 0 ? (
        <div style={{ fontFamily: MONO, fontSize: 12, color: C.darkGray, padding: 40, textAlign: "center", border: `1px solid ${C.border}` }}>
          No skills found. Try a different search or filter.
        </div>
      ) : (
        <div style={{ display: "flex", flexDirection: "column", gap: 6 }}>
          {filtered.map(s => <SkillCard key={s.id} skill={s} onClick={() => setPage({ view: "detail", id: s.id })} />)}
        </div>
      )}
    </section>
  );
}
